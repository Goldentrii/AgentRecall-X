The Loop — Plywood SOP (run 6×)
SOP run_loop(N, plan):
  ASSERT branch == "opt/autonomous-loops"  AND  no_push AND no_publish AND no_version_bump   # REDLINE

  Phase 1 — IMPLEMENT
    impl  = WORKER.implement(plan)                 # small, single-purpose change on branch
    gate  = RUN("npm run build && npm test")
    IF gate.red:
      FIX(once); gate = RUN(...)                    # retry once
      IF still red: REVERT(impl); loop.status = BLOCKED   # carry the failure to the round-table as the finding
    COMMIT(local, "loop N: " + plan)                # never push

  Phase 2 — ROUND-TABLE (parallel, diverse, must TEST not just opine)
    results = PARALLEL[
      TESTER.run(build/test/eval) -> real stdout,
      SKEPTIC.attack(impl)        -> "did it fix the thing or just rename it?",
      RESEARCHER.investigate(loop.theme) -> papers / math / linguistics grounding,
    ]

  Phase 3 — SYNTHESIZE
    synth = CHAIR(results) -> { worked, still_broken, eval_numbers, next_loop_design }

  Phase 4 — ORCHESTRATOR DECISION (me, bird's-eye)
    UPDATE_HTML(loop N)  ;  plywood_feedback(sop_id=946850b5, outcome, note)
    RETURN synth.next_loop_design          # drives loop N+1
